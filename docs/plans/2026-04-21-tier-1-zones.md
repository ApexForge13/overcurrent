// ============================================================================
// ZONE METRICS — what we measure per zone per observation
// ============================================================================

export interface ZoneObservation {
  zoneId: string;
  observedAt: Date;
  tankerCount: number;
  containerShipCount: number;
  bulkCarrierCount: number;
  lngCarrierCount: number;
  totalShipCount: number;
  averageDraft: number | null;       // meters, null if no tankers present
  medianDwellHours: number | null;   // median dwell time of ships currently in zone
  entryCount24h: number;              // ships that entered zone in last 24h
  exitCount24h: number;               // ships that exited zone in last 24h
  metadata?: Record<string, unknown>;
}

// ============================================================================
// BASELINE COMPUTATION
// ============================================================================
//
// For each zone, maintain a rolling 30-day baseline for each metric.
// Baselines become "mature" (usable for triggers) after 30 days of observations.
//
// Store baselines in a ZoneBaseline table, recomputed daily:
//
// model ZoneBaseline {
//   id              String   @id @default(cuid())
//   zoneId          String   // matches MonitoringZone.id
//   metricName      String   // "tankerCount" | "averageDraft" | etc.
//   windowDays      Int      // always 30 for v1
//   mean            Float
//   stddev          Float
//   sampleCount     Int
//   isMature        Boolean  // true when sampleCount >= 30*4 (4 obs/day × 30 days)
//   computedAt      DateTime @default(now())
//
//   @@index([zoneId, metricName])
//   @@index([isMature])
// }
//
// Observation cadence: every 6 hours per zone = 4 obs/day = 120 obs/30 days
// Trigger maturity floor: sampleCount >= 90 (75% coverage over 30d, handles occasional failures)

// ============================================================================
// TRIGGER LOGIC
// ============================================================================
//
// For each zone observation, check each metric against its mature baseline.
// Fire a trigger when: current metric > (baseline.mean + 2 × baseline.stddev)
//                  OR: current metric < (baseline.mean - 2 × baseline.stddev)
//
// Direction mapping per category:
//   crude_export + tanker buildup   → bearish for crude (oversupply, no buyers taking)
//   crude_export + tanker drawdown  → bullish for crude (tight supply)
//   crude_import + tanker buildup   → bullish for crude (demand arriving)
//   crude_import + tanker drawdown  → bearish for crude (demand softening)
//   chokepoint  + anomaly in either direction → high-severity signal regardless of direction
//     (anomalies at chokepoints usually indicate disruption, not supply-demand)
//
// Severity: scaled by z-score, capped at 1.0 at 4 stddev.

export function classifyZoneDirection(
  zone: MonitoringZone,
  metric: 'tankerCount' | 'containerShipCount' | 'bulkCarrierCount' | 'lngCarrierCount',
  deviation: 'above_baseline' | 'below_baseline'
): { direction: -1 | 0 | 1; notes: string } {
  if (zone.category === 'chokepoint') {
    return { direction: 0, notes: 'chokepoint anomaly — direction ambiguous, treat as high-severity alert' };
  }

  const isExport = zone.category.includes('export');
  const isImport = zone.category.includes('import');

  if (isExport) {
    if (deviation === 'above_baseline') return { direction: -1, notes: 'tanker buildup at export terminal → oversupply signal' };
    if (deviation === 'below_baseline') return { direction: 1, notes: 'tanker drawdown at export terminal → tight supply signal' };
  }

  if (isImport) {
    if (deviation === 'above_baseline') return { direction: 1, notes: 'tanker buildup at import terminal → demand arriving' };
    if (deviation === 'below_baseline') return { direction: -1, notes: 'tanker drawdown at import terminal → demand softening' };
  }

  return { direction: 0, notes: 'classification not mapped' };
}
